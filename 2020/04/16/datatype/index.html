二叉树操作集合及习题函数

#include <stdio.h>
#include <stdlib.h>
#define DataType int

typedef struct Node{
	DataType data;
	struct Node *LChild, *RChild;
	//struct Node *Parent;
}BiTree;

void CreateBitree(BiTree **T)//静态生成二叉树
{
	*T = (BiTree *)malloc(sizeof(BiTree));
	(*T)->data = 1;
	(*T)->LChild = (BiTree *)malloc(sizeof(BiTree));
	(*T)->RChild = (BiTree *)malloc(sizeof(BiTree));

	(*T)->LChild->data = 2;
	(*T)->LChild->LChild = (BiTree *)malloc(sizeof(BiTree));
	(*T)->LChild->LChild->data = 4;
	(*T)->LChild->RChild = (*T)->LChild->LChild->LChild = (*T)->LChild->LChild->RChild = NULL;
	
	(*T)->RChild->data = 3;
	(*T)->RChild->LChild = (BiTree *)malloc(sizeof(BiTree));
	(*T)->RChild->LChild->data = 5;
	(*T)->RChild->LChild->LChild = (*T)->RChild->LChild->RChild = NULL;

	(*T)->RChild->RChild = (BiTree *)malloc(sizeof(BiTree));
	(*T)->RChild->RChild->data = 6;
	(*T)->RChild->RChild->LChild = (*T)->RChild->RChild->RChild = NULL;
}

BiTree *InsterBitreeLeftNode(BiTree *T, DataType x)//插入新左子树
{
	if(T == NULL)
		return NULL;
	BiTree *orileft = T->LChild;//先保存原有子树
	BiTree *newleft = (BiTree *)malloc(sizeof(BiTree));
	newleft->data = x;
	newleft->LChild = orileft;
	newleft->RChild = NULL;//创建新子树并赋值
	T->LChild = newleft;//接回原有二叉树
	return T->LChild; 
}

BiTree *InsterBitreeRightNode(BiTree *T, DataType x)//插入新右子树
{
	if(T == NULL)
		return NULL;
	BiTree *oriright = T->RChild;//先保存原有子树
	BiTree *newright = (BiTree *)malloc(sizeof(BiTree));
	newright->data = x;
	newright->RChild = oriright;
	newright->LChild = NULL;//创建新子树并赋值
	T->RChild = newright;//接回原有二叉树
	return T->RChild; 
}

void PreOrderTraverse(BiTree *T)//先序遍历
{
	if(T)
	{
		printf("%d\t",T->data);
		PreOrderTraverse(T->LChild);
		PreOrderTraverse(T->RChild);
	}
	return ;
}

void InOrderTraverse(BiTree *T)//中序遍历
{
	if(T)
	{
		InOrderTraverse(T->LChild);
		printf("%d\t",T->data);
		InOrderTraverse(T->RChild);
	}
	return ;
}

void PostOrderTraverse(BiTree *T)//后序遍历
{
	if(T)
	{
		PostOrderTraverse(T->LChild);
		PostOrderTraverse(T->RChild);
		printf("%d\t",T->data);
	}
	return ;
}

void PrintBiTree(BiTree *T,int inden)
{
	if(T==NULL)
		return ;
	PrintBiTree(T->RChild,inden+1);
	for(int i = 0;i<inden;i++)
		printf("\t");
	if(inden > 0)
		printf("---");
	printf("%d\n",T->data);
	PrintBiTree(T->LChild,inden+1);
}

static int Leafnum = 0;
int LeafNodeSearch(BiTree *T)//寻找叶子节点的数量，递归形式要配合静态变量
{
	if(T)
	{
		if(T->LChild == NULL && T->RChild == NULL)
			Leafnum++;
		LeafNodeSearch(T->LChild);
		LeafNodeSearch(T->RChild);
	}
	else
		return 0;
	return Leafnum;
}

static int Comnum = 1;
int ComTreeDetection(BiTree *T)//检测是否为完全二叉树，未实现
{
	if(T)
	{
		if(T->LChild == NULL && T->RChild != NULL)
			Comnum = 0;
		LeafNodeSearch(T->LChild);
		LeafNodeSearch(T->RChild);
	}
	else
		return 0;
	return Comnum;
}

void Destroy(BiTree **T)//释放二叉树
{
	if((*T) != NULL && (*T)->LChild != NULL)
		Destroy(&(*T)->LChild);
	if((*T) != NULL && (*T)->RChild != NULL)
		Destroy(&(*T)->RChild);
	free(*T);
}

void *DestroyLeftNode(BiTree *T)//删除左子树，因为是递归删除，所以该节点下所有点都会被删除
{
	if(T==NULL || T->LChild==NULL)
		return NULL;
	Destroy(&T->LChild);
	return T;
}

void *DestroyRightNode(BiTree *T)//删除右子树，因为是递归删除，所以该节点下所有点都会被删除
{
	if(T==NULL || T->RChild==NULL)
		return NULL;
	Destroy(&T->RChild);
	return T;
}

}
